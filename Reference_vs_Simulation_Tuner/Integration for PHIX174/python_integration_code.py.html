import pandas as pd
import numpy as np
from scipy import integrate

# Read the CSV file
df = pd.read_csv('/Users/jaren/Downloads/phiX174_std (9).csv', index_col=0)

# Extract temperature and first derivative
T = df['T'].values
dFdT = df['dFdT'].values

# Integrate using cumulative trapezoidal rule
# This recovers the original function F from its derivative dF/dT
F = integrate.cumulative_trapezoid(dFdT, T, initial=0)

# Create output dataframe
result_df = pd.DataFrame({
    'T': T,
    'dFdT': dFdT,
    'F': F
})

# Save to CSV
output_path = '/Users/jaren/Downloads/phiX174_std_integrated.csv'
result_df.to_csv(output_path, index=False)

print(f"Integration complete! Results saved to: {output_path}")
print(f"\nFirst few rows:")
print(result_df.head(10))
